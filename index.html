const axios = require('axios'); module.exports = async function (context, req) { // Die ID wird aus der Query-Parameter oder dem Request-Body gelesen. const id = (req.query.id || (req.body && req.body.id)); // Prüfen, ob die ID vorhanden ist if (!id) { context.res = { status: 400, body: "Bitte eine gültige ID angeben." }; return; } // Die URL der externen API, angepasst mit der ID aus dem Request const apiUrl = `https://api.restful-api.dev/objects/${id}`; try { // Abfrage der externen API const response = await axios.get(apiUrl); // Rückgabe des Ergebnisses als HTTP-Antwort context.res = { status: 200, // OK body: response.data, headers: { 'Content-Type': 'application/json' } }; } catch (error) { // Fehlerbehandlung context.res = { status: error.response ? error.response.status : 500, body: error.message }; } };
